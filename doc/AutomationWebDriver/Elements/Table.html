def body_rows
  begin
    find_element(:tag_name, "tbody").find_elements(:tag_name, "tr").map{ |row| row.child }
  rescue
    nil
  end
end

def body_rows_text
  body_rows.map { |row| row.map { |cell| cell.text } } unless body_rows.nil?
end
