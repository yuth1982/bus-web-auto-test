def headers
  cell_matrix = self.find_elements(:tag_name, "tr").map{ |row| row.child}
  size = cell_matrix.first.select{ |cell| cell.tag_name == "th"}.length
  if size > 1
    cell_matrix.first
  else
    nil
  end
end

def rows
  cell_matrix = self.find_elements(:xpath, ".//tbody/tr").map{ |row| row.child}
  size = cell_matrix.first.select{ |cell| cell.tag_name == "th"}.length
  if size > 1
    cell_matrix[1..-1]
  else
    cell_matrix
  end

end
